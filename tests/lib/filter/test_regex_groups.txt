# Regex group constructs -- filter test data
# Open in mcview (F3), press F6 to open filter, select "Regular expression"

# -------------------------------------------------------------------
# 1. Basic capturing group  (foo|bar)
# Pattern: (error|warning)
# -------------------------------------------------------------------
[2026-04-22 10:01:03] INFO  server started on port 8080
[2026-04-22 10:01:05] ERROR disk full on /dev/sda1
[2026-04-22 10:01:07] WARNING low memory: 128 MB remaining
[2026-04-22 10:01:09] INFO  connection accepted from 192.168.1.10
[2026-04-22 10:01:11] ERROR failed to open /var/log/app.log
[2026-04-22 10:01:13] WARNING retry attempt 3 of 5
[2026-04-22 10:01:15] DEBUG  heartbeat ok

# -------------------------------------------------------------------
# 2. Non-capturing group  (?:foo|bar)
# Pattern: (?:FAIL|PASS): \d+
# -------------------------------------------------------------------
PASS: 142
FAIL: 7
PASS: 99
SKIP: 3
FAIL: 1
PASS: 200

# -------------------------------------------------------------------
# 3. Named capturing group  (?P<name>...)
# Pattern: (?P<ip>\d+\.\d+\.\d+\.\d+)
# -------------------------------------------------------------------
connect from 10.0.0.1 port 22
connect from 192.168.0.55 port 443
refused 172.16.4.200 port 8080
accepted 10.0.0.1 port 22
timeout 8.8.8.8 port 53

# -------------------------------------------------------------------
# 4. Lookahead  (?=...)
# Pattern: \w+(?= failed)
# -------------------------------------------------------------------
auth failed
connect failed
disk ok
write failed
read ok
upload failed

# -------------------------------------------------------------------
# 5. Negative lookahead  (?!...)
# Pattern: user_(?!admin)\w+
# -------------------------------------------------------------------
user_admin logged in
user_alice logged in
user_bob logged in
user_admin privilege escalation
user_carol logged in

# -------------------------------------------------------------------
# 6. Lookbehind  (?<=...)
# Pattern: (?<=exit code )\d+
# -------------------------------------------------------------------
process rsync exited with exit code 0
process ffmpeg exited with exit code 1
process make exited with exit code 2
process test exited with exit code 0
process deploy exited with exit code 127

# -------------------------------------------------------------------
# 7. Negative lookbehind  (?<!...)
# Pattern: (?<!re)load
# -------------------------------------------------------------------
load config from disk
reload nginx config
load balance rules
reload firewall
load kernel module
preload shared libraries

# -------------------------------------------------------------------
# 8. Alternation in group
# Pattern: (Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec) +\d{1,2}
# -------------------------------------------------------------------
Jan  1 00:00:01 host kernel: started
Mar 15 08:30:00 host sshd: accepted
Jul  4 12:00:00 host cron: job fired
Dec 31 23:59:59 host ntp: sync ok
Feb 29 00:00:00 host kernel: leap day

# -------------------------------------------------------------------
# 9. Nested groups
# Pattern: ((\d{1,3}\.){3}\d{1,3}):\d+
# -------------------------------------------------------------------
upstream 10.0.0.1:8080 status 200
upstream 10.0.0.2:8080 status 502
backend 192.168.1.100:3000 timeout
cache 127.0.0.1:6379 hit

# -------------------------------------------------------------------
# 10. Backreference  \1
# Pattern: (\w+) \1
# -------------------------------------------------------------------
the the quick brown fox
error error reading file
this is fine
warning warning disk usage high
hello world
